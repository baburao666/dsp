clc
clear all
t=0:0.001:2;
f=2;
xct=cos(2*pi*f*t);
subplot(2,2,1);

    xnw(i)=sum(xn.*exp(-j*w(i).*n));
end
subplot(2,2,4);

xlabel('\Omega(radian)')
ylabel('F(cos(4\pi n Ts))')

___________________________________________________________3   Reconstruction

clc
clear all
t=0:0.001:2
tt=-2:0.001:2;
F=4
x_t=sin(2*pi*F*t)
x_tt=sinc(2*pi*F*tt)
subplot(2,1,1)
plot(tt,x_tt)
grid on;

hold on
Ts=1/16
ts=0:Ts:2;
x_n_vec=sin(2*pi*F*ts)

n_vec=ts/Ts;

x_r_t=zeros(1,length(t));

for n_indx=1:length(n_vec)
    xn=x_n_vec(1,n_indx);
    n=n_vec(1,n_indx);
    temp=xn*sinc((t-n*Ts)/Ts);
    x_r_t=x_r_t+temp;
end

subplot(2,1,2)
plot(t,x_r_t,'r+:');
grid on;
hold on
subplot(2,1,2)
plot(t,x_t,'k-');
xlabel('t');
legend('Reconstructed signal','Original signal')
